```typescript
// src/app/framework/tokens/popout.token.ts
// VERSION 1 (Section 315) - Pop-out injection token

import { InjectionToken } from '@angular/core';

/**
 * Injection token to signal that the current component context is a pop-out window.
 * Used by services (like ResourceManagementService) to adjust behavior when provided
 * by a pop-out component (e.g., disabling auto-fetch).
 *
 * **Usage:**
 *
 * ```typescript
 * constructor(@Inject(IS_POPOUT_TOKEN) public isPopOut: boolean) {}
 * ```
 *
 * **In Templates:**
 *
 * ```html
 * <nav *ngIf="!isPopOut">Main navigation</nav>
 * <button *ngIf="isPopOut" (click)="close()">Close</button>
 * ```
 *
 * **Note:** This token is provided at the component level, not via a factory.
 * Components that render in pop-out windows provide this token with value `true`.
 *
 * @example
 * ```typescript
 * // In pop-out component
 * @Component({
 *   selector: 'app-panel-popout',
 *   providers: [
 *     { provide: IS_POPOUT_TOKEN, useValue: true }
 *   ]
 * })
 * export class PanelPopoutComponent { }
 *
 * // In main window component
 * @Component({
 *   selector: 'app-discover',
 *   providers: [
 *     { provide: IS_POPOUT_TOKEN, useValue: false }
 *   ]
 * })
 * export class DiscoverComponent { }
 * ```
 */
export const IS_POPOUT_TOKEN = new InjectionToken<boolean>('IS_POPOUT_TOKEN');
```

### Step 315.5: Provide the Token at Component Level

The `IS_POPOUT_TOKEN` is provided at the component level rather than via a factory function. This approach allows explicit control over the token value in different contexts:

```typescript
// In pop-out window component
@Component({
  selector: 'app-panel-popout',
  providers: [
    { provide: IS_POPOUT_TOKEN, useValue: true }
  ]
})
export class PanelPopoutComponent {
  // Services injected here will receive isPopOut = true
}

// In main window component
@Component({
  selector: 'app-discover',
  providers: [
    { provide: IS_POPOUT_TOKEN, useValue: false }
  ]
})
export class DiscoverComponent {
  // Services injected here will receive isPopOut = false
}
```

This pattern is preferred over a factory function because:
1. It provides explicit control over the token value
2. It works reliably across different routing scenarios
3. It avoids relying on URL parsing at bootstrap time
