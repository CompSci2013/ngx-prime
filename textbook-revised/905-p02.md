```typescript
// src/app/app-routing.module.ts
// VERSION 3 - Added 404 wildcard route

import { NgModule } from '@angular/core';
import { RouterModule, Routes } from '@angular/router';

import { HomeComponent } from './features/home/home.component';
import { DiscoverComponent } from './features/discover/discover.component';
import { PanelPopoutComponent } from './features/panel-popout/panel-popout.component';

/**
 * Application Routes Configuration
 *
 * Defines all navigation paths for the Generic-Prime application.
 * Uses direct component references (eager loading) for simplicity.
 *
 * Route Structure:
 * - / (root): Redirects to /home
 * - /home: Home landing page
 * - /discover: Main discovery interface (domain-agnostic)
 * - /panel/:gridId/:panelId/:type: Pop-out window routes
 */
const routes: Routes = [
  // Default route - redirect to home
  { path: '', redirectTo: 'home', pathMatch: 'full' },

  // Home page (eager loaded)
  { path: 'home', component: HomeComponent },

  // Discover page (domain-agnostic discovery interface)
  { path: 'discover', component: DiscoverComponent },

  // Pop-out panel route: /panel/:gridId/:panelId/:type
  // Used by DiscoverComponent.popOutPanel() to open panels in separate windows
  // PanelPopoutComponent handles BroadcastChannel communication with main window
  { path: 'panel/:gridId/:panelId/:type', component: PanelPopoutComponent },

  // 404 fallback route - redirect unmatched paths to home
  { path: '**', redirectTo: 'home' }
];

@NgModule({
  imports: [RouterModule.forRoot(routes)],
  exports: [RouterModule]
})
export class AppRoutingModule {}
```

**Route configuration explained:**

| Route | Purpose | Component |
|-------|---------|-----------|
| `''` | Root path, redirects to home | Immediate redirect |
| `'home'` | Home landing page | HomeComponent |
| `'discover'` | Main discovery interface | DiscoverComponent |
| `'panel/:gridId/:panelId/:type'` | Pop-out windows | PanelPopoutComponent |
| `'**'` | Catch-all for unknown routes | Redirect to home |
